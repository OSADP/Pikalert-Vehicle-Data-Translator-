54 // Structure defining the conditions of the road
55 // Some of these are not used as yet but may be later.
56 struct roadconditions {
57  float road_T; // Pavement temperature (deg C)
58  float subsurface_T; // Subsurface temperature (deg C)
59  int road_condition; // METRo value for road_cond
60  // Dry road = 1
61  // Wet road = 2
62  // Ice/Snow on the road = 3
63  // Mix water/snow on the road = 4
64  // Dew = 5
65  // Melting Snow = 6
66  // Frost = 7
67  // Icing Rain = 8
68  float Tfreezepoint; // Freeze point of the road surface (taking into account applied chemcicals)
69  float PliquidOnRoad; // <reserved> not currently used
70  float PsnowOnRoad; // Snow depth on the road (mm)
71  int roadPtype; // The type of precipiation hitting the road. Modified from wxfcst to
72  // factor in the temperature of the road surface (rain-frz rain)
73  int roadWphase; // The expected phase of the water on the road given current conditions
74 
75  // Variables used in chemical dilution algorithm
76  float AvailH2O; // Water available for chemicals to dissolve into (lbs/sf)
77  float NominalChemConc; // The theoretical chemical concentration (ignoring solution/saturation curve)
78  float AvailableChem; // The amount of pure de-icing chemical available on the road surface.
79  float FinalChemConc; // The final chemical concentration (factoring in dilution and
80  // and solution/saturation curves
81  int ChemSolutionType; // The type of chemical on the road surface, this may change for each hour depending on
82  // the mix of chemicals being used by the DOT. For example, at hour 6 they might put down
83  // salt and at hour 12 MgCl.
84 
85  float Friction; // <reserved> not currently used
86  float NetMobility; // <reserved> not currently used
87 
88  int TrafficIntensity; // Amount of traffic on roadway (light=1 moderate=2 heavy=3)
89 };
