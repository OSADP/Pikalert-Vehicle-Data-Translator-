36 template<typename T>
37 int get_field(const netCDF::NcFile &data_file, const string &field_name, vector<T> &field, const string &units_name, string &units, const string &missing_name, T *missing, string &error)
38 {
39  // Get the variable
40  netCDF::NcVar var;
41  var = data_file.getVar(field_name);
42  if (var.isNull())
43  {
44  error = string("could not find variable: ") + field_name;
45  return -1;
46  }
47 
48  vector<netCDF::NcDim> dims = var.getDims();
49 
50  size_t total_size = 1;
51 
52  for (size_t i=0; i<dims.size(); i++)
53  {
54  total_size *= dims[i].getSize();
55  }
56 
57  if (dims.size() > 0)
58  {
59  field.resize(total_size);
60  var.getVar(&field[0]);
61  }
62  else
63  {
64  field.resize(1);
65  var.getVar(&field[0]);
66  }
67 
68  // Get units if available
69  netCDF::NcVarAtt att;
70 
71  try
72  {
73  att = var.getAtt(units_name);
74  att.getValues(units);
75  }
76  catch (netCDF::exceptions::NcException e)
77  {
78  }
79 
80  try
81  {
82  att = var.getAtt(missing_name);
83  att.getValues(missing);
84  }
85  catch (netCDF::exceptions::NcException e)
86  {
87  }
88 
89  return 0;
90 }
